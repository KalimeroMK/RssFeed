# Site config for pravoslavie.bg
# Bulgarian Orthodox Portal - WordPress site using Hueman theme

# Main content is in .entry-inner inside .entry.themeform
body: //div[@class="entry-inner"]

# Alternative selectors (fallback)
body: //div[contains(@class, "entry-content")]
body: //article//div[contains(@class, "entry")]

# Title extraction
title: //meta[@property="og:title"]/@content
title: //h1[@class="post-title entry-title"]

# Author extraction
author: //meta[@name="author"]/@content
author: //span[@class="authorname"]//a

# Date extraction
date: //meta[@property="article:published_time"]/@content
date: //span[@class="published"]

strip_id_or_class: give-donation-amount
strip_id_or_class: give-donation-levels-wrap

# Related posts section

strip: //div[contains(@class, "related-posts")]

# Comments section

strip: //div[contains(@class, "ad") and contains(@class, "widget")]
